# make graph functions 
def axi_plot(axi, acinst,einst):
    axi.set_title(f'{acinst.metadata["sampleName"]}')
    axi.plot(acinst.df["uvEnergy"],acinst.df['nayield'],'ro-',label=f'Calc\nthreshold:{acinst.metadata["thresholdEnergy"]:.2f}eV\nslope:{acinst.metadata["slope"]:.2f}')
    axi.plot(einst.data_dict["Energy[eV]"],einst.data_dict['Yield^0.5'],'bo-',label=f'Excel\nthreshold:{einst.m_meta_dict["thresholdEnergy"]:.2f}eV\nslope:{einst.m_meta_dict["slope"]:.2f}')

    axi.plot(acinst.df["uvEnergy"],acinst.df["guideline"],'g-',label=f'Estimate line')

    axi.legend()
    axi.grid()
    axi.set_xlabel('Energy [eV]')
    axi.set_ylabel(f'Yield^{acinst.metadata["powerNumber"]:.1f}')
    
    return axi

def axi_diff(axi,acinst,einst,ylim=(-0.06,0.06)):
    
    dfm=np.mean(np.abs(acinst.df['nayield']-np.array(einst.data_dict['Yield^0.5'])))

    axi.set_title(f'{acinst.metadata["sampleName"]}')
    axi.plot(acinst.df["uvEnergy"],acinst.df['nayield']-np.array(einst.data_dict['Yield^0.5']),'ro-',label='dif')
    axi.grid()
    axi.set_xlabel('Energy [eV]')
    axi.set_ylabel(f'Yield^{acinst.metadata["powerNumber"]:.1f} difference (calc - original)')
    axi.legend(title=f'Ave abs dif: {dfm:.4f}') #,loc='upper left'

    axi.set_ylim(*ylim)

    return axi


def make_fig(acinst,einst,ylim,save=False):
    
    width_u = 5.2
    height_u = 4
    nrows=1
    ncols=2
    fig,ax = plt.subplots(nrows=nrows, ncols=ncols, figsize=(ncols*width_u,nrows*height_u), squeeze=False,)
    
    ax[0,0] = axi_plot(ax[0,0],acinst,einst)
    ax[0,1] = axi_diff(ax[0,1],acinst,einst,ylim)
    
    plt.tight_layout()
    
    if save:
        filename=f'{acinst.metadata["sampleName"]}'
        plt.savefig(f'{filename}.png', dpi=300)
    plt.show()
